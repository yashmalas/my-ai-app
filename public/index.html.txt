        <textarea id="prompt" placeholder="Ask me anything..."></textarea>
        <button onclick="askAI()">Send Message</button>
        
        <div id="response"></div>
    </div>

    <script>
        async function askAI() {
            const prompt = document.getElementById('prompt').value;
            const responseDiv = document.getElementById('response');
            
            if (!prompt.trim()) {
                responseDiv.textContent = '⚠️ Please enter something!';
                responseDiv.className = 'show error';
                return;
            }
            
            responseDiv.textContent = '⏳ Thinking...';
            responseDiv.className = 'show loading';
            
            try {
                const response = await fetch('/.netlify/functions/ai-call', {
                    method: 'POST',
                    body: JSON.stringify({ prompt })
                });
                
                const data = await response.json();
                
                if (data.error) {
                    responseDiv.textContent = '❌ Error: ' + data.error;
                    responseDiv.className = 'show error';
                } else {
                    responseDiv.innerHTML = `<strong>AI Says:</strong><p>${data.result}</p>`;
                    responseDiv.className = 'show success';
                    document.getElementById('prompt').value = '';
                }
            } catch (error) {
                responseDiv.textContent = '❌ Error: ' + error.message;
                responseDiv.className = 'show error';
            }
        }
        
        document.getElementById('prompt').addEventListener('keypress', function(event) {
            if (event.key === 'Enter' && !event.shiftKey) {
                event.preventDefault();
                askAI();
            }
        });
    </script>
